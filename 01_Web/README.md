# HTML 기본 태그

```html
<!-- 제목(Heading) -->
<h1>나는 h1!</h1>
<h2>나는 h2!</h2>
<h3>나는 h3!</h3>
<h4>나는 h4!</h4>
<h5>나는 h5!</h5>
<h6>나는 h6!</h6>

<!-- 글씨굵게(Bold) -->
<p><b>저는 굵은 글씨에요.</b></p>
<p><strong>저는 굵은 글씨에요.</strong></p>

<!-- 기울임(Italic) -->
<p><i>기울기울</i></p>
<p><em>저는 강조 의미를 가진 기울임입니다.</em></p> <!-- emphasized-->

<!-- highlighted -->
<p>이 곳은 <mark>멀티캠퍼스 역삼</mark>입니다.</p>

<!-- 취소선(del) / 밑줄(ins) -->
<p>This is <del>del</del></p>
<p>This is <ins>ins</ins></p>

<!-- 아래첨자(sub) / 위첨자(sup) -->
<p>This is <sub>sub</sub></p>
<p>This is <sup>sup</sup></p>

<!-- 단락(p) / 줄바꿈(br) -->
<p>
    This is p!<br>
    This is p!<br>
    This is p!<br>
    This is p!<br>
    This is p!
</p>

<!-- 인용문 꾸미기 -->
<pre>
  from flask import flask
  app = Flask(__name__)
</pre>
<q>
    창오 said, "HTML은 꿀잼."
</q>
<blockquote>
    Hello, HTML!
</blockquote>

<!-- 목록(ol / ul / li) -->
<!-- 순서가 있는 리스트 -->
<ol>
    <li>1</li>
    <li>2</li>
    <li>3</li>
    <li>4</li>
</ol>
<!-- 순서가 없는 리스트 -->
<ul>
    <li>1</li>
    <li>2</li>
    <li>3</li>
    <li>4</li>
</ul>
```
